# =============================================================
# 01. Import count, taxonomy and metadata tables
# =============================================================

# Required packages
library(phyloseq)
library(tidyverse)

# -----------------------------
# Input files
# -----------------------------
# Expected input format:
# 1. abundance/ASV_counts.tsv
#    Rows: taxa/ASV/genera identifiers
#    Columns: sample IDs
#
# 2. taxonomy/taxonomy.tsv
#    Rows: taxa/ASV/genera identifiers
#    Columns: Kingdom, Phylum, Class, Order, Family, Genus, Species
#
# 3. metadata/metadata.csv
#    Rows: sample IDs
#    Columns: clinical/demographic variables used in the manuscript

otu_file  <- "../abundance/ASV_counts.tsv"
tax_file  <- "../taxonomy/taxonomy.tsv"
meta_file <- "../metadata/metadata.csv"

# -----------------------------
# Import tables
# -----------------------------
otu <- read.delim(
  otu_file,
  row.names = 1,
  check.names = FALSE,
  sep = "\t"
)

tax <- read.delim(
  tax_file,
  row.names = 1,
  check.names = FALSE,
  sep = "\t"
)

meta <- read.csv(
  meta_file,
  row.names = 1,
  check.names = FALSE
)

# -----------------------------
# Basic checks
# -----------------------------
common_samples <- intersect(colnames(otu), rownames(meta))

if (length(common_samples) == 0) {
  stop("No matching sample IDs found between abundance table and metadata.")
}

otu <- otu[, common_samples]
meta <- meta[common_samples, , drop = FALSE]

common_taxa <- intersect(rownames(otu), rownames(tax))

if (length(common_taxa) == 0) {
  stop("No matching taxa IDs found between abundance and taxonomy tables.")
}

otu <- otu[common_taxa, , drop = FALSE]
tax <- tax[common_taxa, , drop = FALSE]

# Ensure numeric abundance table
otu <- as.matrix(otu)
mode(otu) <- "numeric"

# -----------------------------
# Create phyloseq object
# -----------------------------
OTU <- otu_table(otu, taxa_are_rows = TRUE)
TAX <- tax_table(as.matrix(tax))
SAM <- sample_data(meta)

ps <- phyloseq(OTU, TAX, SAM)

# Remove taxa with zero counts across all samples
ps <- prune_taxa(taxa_sums(ps) > 0, ps)

# Save object
saveRDS(ps, "../results/phyloseq_object_raw_counts.rds")

# Export summary
summary_table <- data.frame(
  n_samples = nsamples(ps),
  n_taxa = ntaxa(ps),
  total_reads = sum(sample_sums(ps))
)

write.csv(summary_table, "../results/import_summary.csv", row.names = FALSE)

cat("Phyloseq object created successfully.\n")
cat("Samples:", nsamples(ps), "\n")
cat("Taxa:", ntaxa(ps), "\n")
cat("Total reads:", sum(sample_sums(ps)), "\n")
EOF

# =============================================================
# 02. Relative abundance calculation and taxonomic aggregation
# =============================================================

library(phyloseq)
library(microbiome)
library(tidyverse)

ps <- readRDS("../results/phyloseq_object_raw_counts.rds")

# -----------------------------
# Transform to relative abundance
# -----------------------------
ps_rel <- microbiome::transform(ps, transform = "compositional")

saveRDS(ps_rel, "../results/phyloseq_object_relative_abundance.rds")

# -----------------------------
# Aggregate to selected taxonomic levels
# -----------------------------
tax_levels <- c("Phylum", "Family", "Genus")

for (level in tax_levels) {

  if (!level %in% rank_names(ps_rel)) {
    warning(paste("Taxonomic level not found:", level))
    next
  }

  ps_level <- tax_glom(ps_rel, taxrank = level, NArm = FALSE)

  abundance_df <- psmelt(ps_level) %>%
    group_by(Sample, !!sym(level)) %>%
    summarise(Abundance = sum(Abundance), .groups = "drop")

  out_file <- paste0("../results/relative_abundance_", level, ".csv")
  write.csv(abundance_df, out_file, row.names = FALSE)
}

# -----------------------------
# Prepare summarized abundance tables for plotting
# -----------------------------

# Phylum: retain taxa >1% mean relative abundance
if ("Phylum" %in% rank_names(ps_rel)) {

  ps_phylum <- tax_glom(ps_rel, "Phylum", NArm = FALSE)
  df_phylum <- psmelt(ps_phylum)

  phylum_keep <- df_phylum %>%
    group_by(Phylum) %>%
    summarise(mean_abundance = mean(Abundance), .groups = "drop") %>%
    filter(mean_abundance > 0.01) %>%
    pull(Phylum)

  df_phylum <- df_phylum %>%
    mutate(Phylum_plot = ifelse(Phylum %in% phylum_keep, Phylum, "Other"))

  write.csv(df_phylum, "../results/plot_table_phylum_abundance.csv", row.names = FALSE)
}

# Family: retain taxa >5% mean relative abundance
if ("Family" %in% rank_names(ps_rel)) {

  ps_family <- tax_glom(ps_rel, "Family", NArm = FALSE)
  df_family <- psmelt(ps_family)

  family_keep <- df_family %>%
    group_by(Family) %>%
    summarise(mean_abundance = mean(Abundance), .groups = "drop") %>%
    filter(mean_abundance > 0.05) %>%
    pull(Family)

  df_family <- df_family %>%
    mutate(Family_plot = ifelse(Family %in% family_keep, Family, "Other"))

  write.csv(df_family, "../results/plot_table_family_abundance.csv", row.names = FALSE)
}

# Genus: retain top 20 genera
if ("Genus" %in% rank_names(ps_rel)) {

  ps_genus <- tax_glom(ps_rel, "Genus", NArm = FALSE)
  df_genus <- psmelt(ps_genus)

  genus_keep <- df_genus %>%
    group_by(Genus) %>%
    summarise(mean_abundance = mean(Abundance), .groups = "drop") %>%
    arrange(desc(mean_abundance)) %>%
    slice_head(n = 20) %>%
    pull(Genus)

  df_genus <- df_genus %>%
    mutate(Genus_plot = ifelse(Genus %in% genus_keep, Genus, "Other"))

  write.csv(df_genus, "../results/plot_table_genus_top20_abundance.csv", row.names = FALSE)
}

cat("Relative abundance analysis completed.\n")
EOF

# -----------------------------
# Function for statistical tests
# -----------------------------
run_alpha_tests <- function(data, variable, indices = c("ACE", "Shannon", "Simpson")) {

  variable <- as.character(variable)

  if (!variable %in% colnames(data)) {
    warning(paste("Variable not found:", variable))
    return(NULL)
  }

  results <- list()

  for (idx in indices) {

    df <- data %>%
      select(all_of(variable), all_of(idx)) %>%
      drop_na()

    colnames(df) <- c("Group", "Value")
    df$Group <- as.factor(df$Group)

    n_groups <- nlevels(df$Group)

    if (n_groups < 2) {
      next
    }

    # Normality by group
    shapiro_p <- df %>%
      group_by(Group) %>%
      summarise(
        shapiro_p = ifelse(n() >= 3, shapiro.test(Value)$p.value, NA_real_),
        .groups = "drop"
      )

    normal_data <- all(shapiro_p$shapiro_p > 0.05, na.rm = TRUE)

    if (n_groups == 2) {
      if (normal_data) {
        test <- t.test(Value ~ Group, data = df)
        test_name <- "Student t-test"
        statistic <- unname(test$statistic)
      } else {
        test <- wilcox.test(Value ~ Group, data = df, exact = FALSE)
        test_name <- "Wilcoxon rank-sum test"
        statistic <- unname(test$statistic)
      }
    } else {
      if (normal_data) {
        test <- aov(Value ~ Group, data = df)
        test_summary <- summary(test)[[1]]
        test_name <- "ANOVA"
        statistic <- test_summary$`F value`[1]
        pvalue <- test_summary$`Pr(>F)`[1]
      } else {
        test <- kruskal.test(Value ~ Group, data = df)
        test_name <- "Kruskal-Wallis test"
        statistic <- unname(test$statistic)
        pvalue <- test$p.value
      }
    }

    if (!exists("pvalue")) {
      pvalue <- test$p.value
    }

    results[[paste(variable, idx, sep = "_")]] <- data.frame(
      Variable = variable,
      Index = idx,
      Test = test_name,
      Statistic = statistic,
      P_value = pvalue
    )

    rm(pvalue)
  }

  bind_rows(results)
}

# -----------------------------
# PERMANOVA function
# -----------------------------
run_permanova <- function(ps_object, distance_object, variable, permutations = 999) {

  metadata <- as(sample_data(ps_object), "data.frame")

  if (!variable %in% colnames(metadata)) {
    warning(paste("Variable not found:", variable))
    return(NULL)
  }

  metadata <- metadata[labels(distance_object), , drop = FALSE]

  formula_text <- paste("distance_object ~", variable)
  formula_obj <- as.formula(formula_text)

  ad <- adonis2(
    formula_obj,
    data = metadata,
    permutations = permutations
  )

  data.frame(
    Variable = variable,
    R2 = ad$R2[1],
    F_statistic = ad$F[1],
    P_value = ad$`Pr(>F)`[1]
  )
}

# -----------------------------
# Function to run ANCOM-BC2 for one variable
# -----------------------------
run_ancombc2 <- function(ps_object, variable) {

  metadata <- as(sample_data(ps_object), "data.frame")

  if (!variable %in% colnames(metadata)) {
    warning(paste("Variable not found:", variable))
    return(NULL)
  }

  metadata[[variable]] <- as.factor(metadata[[variable]])
  sample_data(ps_object) <- sample_data(metadata)

  if (nlevels(metadata[[variable]]) < 2) {
    warning(paste("Variable has fewer than 2 levels:", variable))
    return(NULL)
  }

  set.seed(123)

  output <- ancombc2(
    data = ps_object,
    assay_name = "counts",
    tax_level = "Genus",
    fix_formula = variable,
    rand_formula = NULL,
    p_adj_method = "BH",
    pseudo = 0,
    pseudo_sens = TRUE,
    prv_cut = 0.10,
    lib_cut = 0,
    s0_perc = 0.05,
    group = variable,
    struc_zero = TRUE,
    neg_lb = TRUE,
    alpha = 0.05,
    n_cl = 1,
    verbose = TRUE,
    global = TRUE,
    pairwise = TRUE,
    dunnet = FALSE,
    trend = FALSE
  )

  res <- output$res

  # Export all available ANCOMBC2 result tables
  out_prefix <- paste0("../results/ANCOMBC2_", variable)

  if (!is.null(res$lfc)) {
    write.csv(res$lfc, paste0(out_prefix, "_lfc.csv"))
  }

  if (!is.null(res$se)) {
    write.csv(res$se, paste0(out_prefix, "_se.csv"))
  }

  if (!is.null(res$p_val)) {
    write.csv(res$p_val, paste0(out_prefix, "_pvalues.csv"))
  }

  if (!is.null(res$q_val)) {
    write.csv(res$q_val, paste0(out_prefix, "_qvalues.csv"))
  }

  if (!is.null(res$diff_abn)) {
    write.csv(res$diff_abn, paste0(out_prefix, "_diff_abundance.csv"))
  }

  return(output)
}

# -----------------------------
# Function to run DESeq2 for one variable
# -----------------------------
run_deseq2 <- function(ps_object, variable) {

  metadata <- as(sample_data(ps_object), "data.frame")

  if (!variable %in% colnames(metadata)) {
    warning(paste("Variable not found:", variable))
    return(NULL)
  }

  metadata[[variable]] <- as.factor(metadata[[variable]])

  if (nlevels(metadata[[variable]]) < 2) {
    warning(paste("Variable has fewer than 2 levels:", variable))
    return(NULL)
  }

  sample_data(ps_object) <- sample_data(metadata)

  design_formula <- as.formula(paste("~", variable))

  dds <- phyloseq_to_deseq2(ps_object, design_formula)

  # Geometric mean replacement for sparse microbiome data
  gm_mean <- function(x, na.rm = TRUE) {
    exp(sum(log(x[x > 0]), na.rm = na.rm) / length(x))
  }

  geo_means <- apply(counts(dds), 1, gm_mean)
  dds <- estimateSizeFactors(dds, geoMeans = geo_means)

  dds <- DESeq(dds, test = "Wald", fitType = "parametric")

  result_names <- resultsNames(dds)
  result_names <- result_names[result_names != "Intercept"]

  all_results <- list()

  tax_df <- as.data.frame(tax_table(ps_object)) %>%
    rownames_to_column("TaxonID")

  for (coef_name in result_names) {

    res <- results(dds, name = coef_name, alpha = 0.01)
    res_df <- as.data.frame(res) %>%
      rownames_to_column("TaxonID") %>%
      left_join(tax_df, by = "TaxonID") %>%
      arrange(padj)

    out_file <- paste0("../results/DESeq2_", variable, "_", coef_name, ".csv")
    write.csv(res_df, out_file, row.names = FALSE)

    sig <- res_df %>%
      filter(!is.na(padj), padj < 0.01)

    write.csv(sig, sig_file, row.names = FALSE)

    all_results[[coef_name]] <- res_df
  }

  return(all_results)
}

plot_stacked_bar <- function(df, x_variable, fill_variable, output_file) {

  if (!x_variable %in% colnames(df)) {
    warning(paste("Variable not found:", x_variable))
    return(NULL)
  }

  p <- ggplot(df, aes_string(x = x_variable, y = "Abundance", fill = fill_variable)) +
    geom_bar(stat = "identity", position = "fill") +
    theme_bw() +
    labs(x = x_variable, y = "Relative abundance", fill = fill_variable) +
    theme(
      axis.text.x = element_text(angle = 45, hjust = 1),
      panel.grid.major.x = element_blank()
    )

  ggsave(output_file, p, width = 9, height = 6, dpi = 300)

  return(p)
}
